CREATE TABLE studenti(
    matricola INT NOT NULL PRIMARY KEY,
    nome VARCHAR(100) NOT NULL, 
    cognome VARCHAR(100) NOT NULL,
    eta INT NOT NULL,
    nascita DATE,
  	UNIQUE (nome, cognome),
    CHECK(eta >= 18)
);
